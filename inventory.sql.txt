-- Create the Inventory table
CREATE TABLE Inventory (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    ProductName VARCHAR(255) NOT NULL,
    ModelName VARCHAR(255) NOT NULL,
    StockCount INT NOT NULL,
    ProductPrice DOUBLE NOT NULL
);
